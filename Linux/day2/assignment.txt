# ==========================================
#      Day 2: User and Group Management
# ==========================================

# Part A: User Creation and Management
# ------------------------------------------

# 1. Create three new users
sudo adduser alice                             # Interactive creation
sudo adduser --home /opt/users/bob bob         # Custom home directory
sudo adduser --system charlie                  # System user

# 2. Set passwords for alice and bob
sudo passwd alice
sudo passwd bob
sudo passwd -e alice                           # Force password change at next login

# 3. View user information
grep alice /etc/passwd                         # Display alice's entry
sudo chage -l bob                              # Show bob's password aging info


# Part B: Group Management
# ------------------------------------------

# 4. Create two groups
sudo addgroup developers
sudo addgroup project-team

# 5. Add users to groups
sudo usermod -aG developers,project-team alice # Add alice to both groups
sudo usermod -aG developers bob                # Add bob to developers
sudo usermod -g developers alice               # Make developers alice's primary group

# 6. Verify group memberships
groups alice                                   # Show alice's groups
id bob                                         # Display bob's UID and GID


# Part C: File Permissions and Ownership
# ------------------------------------------

# 7. Create test files and directories
mkdir /tmp/lab1_test
echo "Important project data" > /tmp/lab1_test/data.txt
touch /tmp/lab1_test/script.sh

# 8. Set permissions using symbolic notation
chmod u=rw,g=r,o=r /tmp/lab1_test/data.txt     # rw-r--r--
chmod u=rwx,go=rx /tmp/lab1_test/script.sh     # rwxr-xr-x

# 9. Change ownership
sudo chown alice:developers /tmp/lab1_test/data.txt
# Note: Executing recursive chown on the directory after the file will overwrite the file's ownership.
sudo chown -R bob:project-team /tmp/lab1_test


# Part D: Verification and Testing
# ------------------------------------------

# 10. Test permissions (Switching user)
sudo su - alice
cat /tmp/lab1_test/data.txt                    # Alice can read (Others have read permission)
/tmp/lab1_test/script.sh                       # Alice can execute (Others have execute permission)
exit                                           # Return to moaz

# 11. Display final state
ls -la /tmp/lab1_test                          # Show permissions and ownership
groups alice bob charlie                       # Show group memberships for all created users
