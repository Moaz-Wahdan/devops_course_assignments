# Task 1: Create a directory structure for a project
# ------------------------------------------

# 1. Create the main project directory
mkdir my_project

# 2. Navigate into the newly created project directory
cd my_project

# 3. Create the standard subdirectories (src, docs, tests)
mkdir src docs tests

# 4. Create the specified files inside their respective directories
touch docs/syteminfo
touch tests/dummy-data
touch README.txt

# Verify the structure was created successfully (requires 'tree' command to be installed) Or simply use ls -R
ls -R



# Task 2: Practice all the commands learned today
# ------------------------------------------

# Navigation & Directory Operations
pwd                          # Print the absolute path of the current working directory
ls -la                       # List all files (including hidden) in detailed long format
cd ~                         # Return to the user's home directory
cd -                         # Switch back to the previous directory

# File Creation & Manipulation
touch practice_file.txt                     # Create an empty text file
echo "Hello DevOps" > practice_file.txt     # Write initial text to the file
echo "Learning Linux" >> practice_file.txt  # Append new text without overwriting
cp practice_file.txt backup_file.txt        # Copy the file to a new destination
mv backup_file.txt old_backup.txt           # Rename the copied file
rm old_backup.txt                           # Delete the renamed file permanently

# Viewing File Contents
cat practice_file.txt        # Display the entire content of the file
head -n 1 practice_file.txt  # Display only the first line of the file
tail -n 1 practice_file.txt  # Display only the last line of the file

# System Information Exploration
uname -a                     # Print comprehensive system information (kernel, OS, architecture)
whoami                       # Display the currently logged-in username
df -h                        # Check disk space usage in a human-readable format
du -sh *                     # Calculate and display the size of files/folders in the current directory



# Task 3: Read the manual page for 'ls' command
# ------------------------------------------

# Open the manual documentation for 'ls' to explore available flags and options
man ls


# Task 4: Explore the system's /etc directory (read-only)
# ------------------------------------------

# Navigate to the /etc directory where system-wide configuration files are stored
cd /etc

# List all configuration files and directories to explore the structure
ls -la

# Safely view the contents of a specific configuration file (e.g., OS release info)
cat os-release
